1. npm init -y
2. npm i typescript ts-node @types/node nodemon --save-dev 
3. npx tsc --init

4. npm i jest @types/jest ts-jest supertest @types/supertest jest-environment-node jest-config --save-dev 
5. npm i express dotenv cors helmet morgan jest-config 
6. npm i @types/express @types/cors @types/morgan --save-dev 
7. npm i bcrypt compression express-rate-limit express-validator jsonwebtoken multer 
8. npm i @types/bcrypt @types/compression @types/jsonwebtoken @types/multer --save-dev

9. npm i prisma @types/node @types/pg --save-dev    
10. npm i @prisma/client @prisma/adapter-pg pg dotenv 
11. npx prisma init --db --output ../generated/prisma 
12. npx prisma generate 
13. npx prisma migrate dev --name init 

14. npx prisma migrate dev // migrate data
15. npx prisma db push // direct sync with database 
16. npx prisma migrate dev --create-only // only create migration file
17. npx prisma migrate reset

18. npx prisma db seed (or) npx tsx prisma/seed.ts
19. npx prisma db pull 
20. npx prisma migrate diff \ 
    --from-empty \
    --to-schema prisma/schema.prisma \
    --script > prisma/migrations/0_init/migration.sql 

21. npx prisma migrate resolve --applied 0_init
22. npx prisma migrate deploy 

23. npm i ejs
